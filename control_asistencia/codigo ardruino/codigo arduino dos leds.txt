#include <SPI.h>
#include <MFRC522.h>

#define SS_PIN 10
#define RST_PIN 9
#define LED_VERDE 7
#define LED_ROJO 6

MFRC522 mfrc522(SS_PIN, RST_PIN);

void setup() {
  Serial.begin(9600);
  
  // Esperar conexión serial con timeout
  unsigned long startTime = millis();
  while (!Serial && (millis() - startTime) < 5000) {
    delay(100);
  }
  
  SPI.begin();
  
  // RESET COMPLETO del RC522
  mfrc522.PCD_Init();
  delay(100);
  mfrc522.PCD_Init(); // Doble init para asegurar
  
  pinMode(LED_VERDE, OUTPUT);
  pinMode(LED_ROJO, OUTPUT);
  
  // Secuencia de reset
  for(int i = 0; i < 5; i++) {
    digitalWrite(LED_VERDE, !digitalRead(LED_VERDE));
    digitalWrite(LED_ROJO, !digitalRead(LED_ROJO));
    delay(200);
  }
  
  digitalWrite(LED_VERDE, LOW);
  digitalWrite(LED_ROJO, LOW);
  
  Serial.println("ARDUINO_RESET_OK");
}

void loop() {
  // Reset periódico del RC522 cada 2 minutos
  static unsigned long lastReset = 0;
  if (millis() - lastReset > 120000) {
    mfrc522.PCD_Init();
    lastReset = millis();
    Serial.println("RC522_RESET");
  }
  
  if (mfrc522.PICC_IsNewCardPresent() && mfrc522.PICC_ReadCardSerial()) {
    String uid = leerUID();
    
    Serial.println("{\"uid\":\"" + uid + "\"}");
    
    digitalWrite(LED_VERDE, HIGH);
    delay(300);
    digitalWrite(LED_VERDE, LOW);
    
    delay(1000);
  }
  
  delay(100);
}

String leerUID() {
  String uid = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if(mfrc522.uid.uidByte[i] < 0x10) uid += "0";
    uid += String(mfrc522.uid.uidByte[i], HEX);
  }
  uid.toUpperCase();
  return uid;
}
